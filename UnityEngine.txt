using System.Collections;
using UnityEngine;

public class SphereCreator : MonoBehaviour
{
    public int numberOfSpheres = 10; // Set the desired number of spheres

    public GameObject spherePrefab; // Reference to the sphere prefab

    void Start()
    {
        CreateSpheres();
    }

    void CreateSpheres()
    {
        for (int i = 0; i < numberOfSpheres; i++)
        {
            // Generate a random position within a range
            float randomX = Random.Range(0f, 3f);
            float randomY = Random.Range(0f, 3f);
            Vector3 randomPosition = new Vector3(randomX, randomY, 0f);

            // Instantiate the spherePrefab at the random position
            GameObject sphere = Instantiate(spherePrefab, randomPosition, Quaternion.identity);

            // Set the material color to red, green, or blue randomly
            Material sphereMaterial = sphere.GetComponent<Renderer>().material;
            sphereMaterial.color = Random.ColorHSV();
        }
    }
}
